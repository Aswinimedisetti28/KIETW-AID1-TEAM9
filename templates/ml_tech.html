import torch
import torch.nn as nn
import torch.nn.functional as F
class EmotionModel(nn.Module):
def __init__(self):
super().__init__()
# Text encoder (BERT based)
self.text_encoder = TextEncoder()
# Voice feature extractor
self.voice_cnn = VoiceCNN()
# Face emotion detector
self.face_net = FaceNet()
# Fusion layer
self.fusion = nn.Linear(768, 256)
self.classifier = nn.Linear(256, 8)
def forward(self, text, voice, face):
text_features = self.text_encoder(text)
voice_features = self.voice_cnn(voice)
face_features = self.face_net(face)
# Multi-modal fusion
combined = torch.cat([text_features, voice_features, face_features], dim=1)
fused = F.relu(self.fusion(combined))
output = self.classifier(fused)
return F.softmax(output, dim=1)
